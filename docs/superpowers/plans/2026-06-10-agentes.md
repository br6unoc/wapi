# Agentes — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Adicionar agentes de IA conversacionais que respondem automaticamente mensagens WhatsApp, com handoff manual e automático para atendentes humanos.

**Architecture:** Cada agente é vinculado a uma conexão (instance) e um tipo de contato (`first_contact` / `returning`). Ao chegar mensagem de texto, `saveMessageAndReply` salva no DB e chama `tryAgentReply` na mesma goroutine; `tryAgentReply` verifica se o agente está ativo, carrega o histórico e chama o provider de AI (OpenAI ou Groq). Handoff ocorre quando: agente detecta keyword configurada na resposta, atendente envia mensagem pela UI, ou atendente clica em "Assumir conversa".

**Tech Stack:** Go 1.24, Gin, PostgreSQL, whatsmeow, Alpine.js, Tailwind CSS, OpenAI/Groq API (HTTP puro, sem SDK)

**Módulo Go:** `botwapp`

**Container de dev:** `wapi-dev` — build: `docker exec wapi-dev sh -c "cd /app && go build ./..."`; restart: `docker exec wapi-dev kill 1` (processo principal reinicia via entrypoint)

## Task 2: Pacote AI — wrapper OpenAI/Groq

**Files:**

- [ ] **Step 1: Criar o arquivo**

Crie `internal/aiagent/aiagent.go` com o conteúdo abaixo. O pacote usa `net/http` puro — sem SDK externo. Groq e OpenAI usam formato idêntico de request/response.

```go
package aiagent

import (
	"bytes"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"time"
)

type Config struct {
	Provider string // "openai" | "groq"
	Model    string
	APIKey   string
}

type Message struct {
	Role    string // "user" | "assistant"
	Content string
}

func Chat(cfg Config, systemPrompt string, history []Message) (string, error) {
	endpoint := "https://api.openai.com/v1/chat/completions"
	if cfg.Provider == "groq" {
		endpoint = "https://api.groq.com/openai/v1/chat/completions"
	}

	msgs := make([]map[string]string, 0, len(history)+1)
	msgs = append(msgs, map[string]string{"role": "system", "content": systemPrompt})
	for _, m := range history {
		msgs = append(msgs, map[string]string{"role": m.Role, "content": m.Content})
	}

	payload := map[string]interface{}{
		"model":       cfg.Model,
		"messages":    msgs,
		"temperature": 0.7,
	}
	body, _ := json.Marshal(payload)

	req, err := http.NewRequest("POST", endpoint, bytes.NewReader(body))
	if err != nil {
		return "", err
	}
	req.Header.Set("Content-Type", "application/json")
	req.Header.Set("Authorization", "Bearer "+cfg.APIKey)

	client := &http.Client{Timeout: 30 * time.Second}
	resp, err := client.Do(req)
	if err != nil {
		return "", fmt.Errorf("chamada AI falhou: %w", err)
	}
	defer resp.Body.Close()

	data, err := io.ReadAll(resp.Body)
	if err != nil {
		return "", err
	}
	if resp.StatusCode != http.StatusOK {
		return "", fmt.Errorf("AI API retornou %d: %s", resp.StatusCode, string(data))
	}

	var result struct {
		Choices []struct {
			Message struct {
				Content string `json:"content"`
			} `json:"message"`
		} `json:"choices"`
	}
	if err := json.Unmarshal(data, &result); err != nil {
		return "", fmt.Errorf("parse resposta AI: %w", err)
	}
	if len(result.Choices) == 0 {
		return "", fmt.Errorf("AI retornou resposta vazia")
	}
	return result.Choices[0].Message.Content, nil
}
```

```go
package handler

import (
	"net/http"

	"github.com/google/uuid"

type agentRow struct {
	ID             string `json:"id"`
	Name           string `json:"name"`
	Prompt         string `json:"prompt"`
	ContactType    string `json:"contact_type"`
	InstanceID     string `json:"instance_id"`
	InstanceName   string `json:"instance_name"`
	IsActive       bool   `json:"is_active"`
	HandoffKeyword string `json:"handoff_keyword"`
}

func ListAgents(c *gin.Context) {
	companyID := currentCompanyID(c)
	rows, err := postgres.DB.Query(`
		SELECT a.id, a.name, a.prompt, a.contact_type, a.instance_id,
		       i.name AS instance_name, a.is_active, a.handoff_keyword
		FROM agents a
		JOIN instances i ON i.id = a.instance_id
		WHERE a.company_id = $1
		ORDER BY i.name, a.contact_type
	`, companyID)
	if err != nil {
		c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
		return
	}
	defer rows.Close()

	agents := make([]agentRow, 0)
	for rows.Next() {
		var a agentRow
		if err := rows.Scan(&a.ID, &a.Name, &a.Prompt, &a.ContactType,
			&a.InstanceID, &a.InstanceName, &a.IsActive, &a.HandoffKeyword); err != nil {
			continue
		}
		agents = append(agents, a)
	}
	c.JSON(http.StatusOK, agents)
}

func CreateAgent(c *gin.Context) {
	companyID := currentCompanyID(c)
	var req struct {
		Name           string `json:"name" binding:"required"`
		Prompt         string `json:"prompt" binding:"required"`
		ContactType    string `json:"contact_type" binding:"required"`
		InstanceID     string `json:"instance_id" binding:"required"`
		IsActive       bool   `json:"is_active"`
		HandoffKeyword string `json:"handoff_keyword"`
	}
	if err := c.ShouldBindJSON(&req); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": "campos obrigatórios: name, prompt, contact_type, instance_id"})
		return
	}
	if req.ContactType != "first_contact" && req.ContactType != "returning" {
		c.JSON(http.StatusBadRequest, gin.H{"error": "contact_type deve ser 'first_contact' ou 'returning'"})
		return
	}
	if req.HandoffKeyword == "" {
		req.HandoffKeyword = "PRECISO_DE_HUMANO"
	}

	id := uuid.New().String()
	_, err := postgres.DB.Exec(`
		INSERT INTO agents (id, company_id, instance_id, name, prompt, contact_type, is_active, handoff_keyword)
		VALUES ($1, $2, $3, $4, $5, $6, $7, $8)
	`, id, companyID, req.InstanceID, req.Name, req.Prompt, req.ContactType, req.IsActive, req.HandoffKeyword)
	if err != nil {
		c.JSON(http.StatusConflict, gin.H{"error": "já existe um agente deste tipo para esta conexão"})
		return
	}

	var instanceName string
	postgres.DB.QueryRow(`SELECT name FROM instances WHERE id = $1`, req.InstanceID).Scan(&instanceName)

	c.JSON(http.StatusCreated, agentRow{
		ID: id, Name: req.Name, Prompt: req.Prompt, ContactType: req.ContactType,
		InstanceID: req.InstanceID, InstanceName: instanceName,
		IsActive: req.IsActive, HandoffKeyword: req.HandoffKeyword,
	})
}

func UpdateAgent(c *gin.Context) {
	id := c.Param("id")
	companyID := currentCompanyID(c)
	var req struct {
		Name           string `json:"name" binding:"required"`
		Prompt         string `json:"prompt" binding:"required"`
		ContactType    string `json:"contact_type" binding:"required"`
		InstanceID     string `json:"instance_id" binding:"required"`
		IsActive       bool   `json:"is_active"`
		HandoffKeyword string `json:"handoff_keyword"`
	}
	if err := c.ShouldBindJSON(&req); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": "campos obrigatórios faltando"})
		return
	}
	if req.ContactType != "first_contact" && req.ContactType != "returning" {
		c.JSON(http.StatusBadRequest, gin.H{"error": "contact_type inválido"})
		return
	}
	if req.HandoffKeyword == "" {
		req.HandoffKeyword = "PRECISO_DE_HUMANO"
	}

	res, err := postgres.DB.Exec(`
		UPDATE agents
		SET name = $1, prompt = $2, contact_type = $3, instance_id = $4,
		    is_active = $5, handoff_keyword = $6
		WHERE id = $7 AND company_id = $8
	`, req.Name, req.Prompt, req.ContactType, req.InstanceID,
		req.IsActive, req.HandoffKeyword, id, companyID)
	if err != nil {
		c.JSON(http.StatusConflict, gin.H{"error": "já existe um agente deste tipo para esta conexão"})
		return
	}
	n, _ := res.RowsAffected()
	if n == 0 {
		c.JSON(http.StatusNotFound, gin.H{"error": "agente não encontrado"})
		return
	}
	c.JSON(http.StatusOK, gin.H{"message": "agente atualizado"})
}

func DeleteAgent(c *gin.Context) {
	id := c.Param("id")
	companyID := currentCompanyID(c)
	_, err := postgres.DB.Exec(`DELETE FROM agents WHERE id = $1 AND company_id = $2`, id, companyID)
	if err != nil {
		c.JSON(http.StatusInternalServerError, gin.H{"error": "erro ao remover agente"})
		return
	}
	c.JSON(http.StatusOK, gin.H{"message": "agente removido"})
}
```

```go
// saveMessageAndReply persiste mensagem de texto recebida e aciona o agente se ativo.
// Substitui `go saveMessage(...)` para mensagens de texto de contatos diretos.
func (inst *Instance) saveMessageAndReply(phone, pushName, content, waID string) {
	var contactID string
	err := postgres.DB.QueryRow(
		`INSERT INTO contacts (instance_id, phone, name)
		 VALUES ($1, $2, $3)
		 ON CONFLICT (instance_id, phone) DO UPDATE SET name = EXCLUDED.name
		 RETURNING id`,
		inst.ID, phone, pushName,
	).Scan(&contactID)
	if err != nil {
		log.Printf("[DB] Erro ao upsert contact: %v", err)
		return
	}

	msgID := uuid.New().String()
	var createdAt string
	err = postgres.DB.QueryRow(
		`INSERT INTO messages (id, instance_id, contact_id, direction, content, type, wa_message_id, media_path, media_name)
		 VALUES ($1, $2, $3, 'in', $4, 'text', $5, '', '')
		 RETURNING created_at`,
		msgID, inst.ID, contactID, content, waID,
	).Scan(&createdAt)
	if err != nil {
		log.Printf("[DB] Erro ao salvar mensagem: %v", err)
		return
	}

	postgres.DB.Exec(`UPDATE contacts SET unread_count = unread_count + 1 WHERE id = $1`, contactID)

	payload := map[string]interface{}{
		"event": "new_message",
		"data": map[string]interface{}{
			"id":            msgID,
			"instance_id":   inst.ID,
			"instance_name": inst.Name,
			"contact_id":    contactID,
			"phone":         phone,
			"name":          pushName,
			"direction":     "in",
			"content":       content,
			"type":          "text",
			"media_path":    "",
			"media_name":    "",
			"created_at":    createdAt,
		},
	}
	jsonBytes, _ := json.Marshal(payload)
	hub.Global.BroadcastToCompany(inst.CompanyID, jsonBytes)

	inst.tryAgentReply(contactID, phone)
}

// tryAgentReply verifica se há agente ativo para o contato e, se sim, chama a AI e envia resposta.
func (inst *Instance) tryAgentReply(contactID, phone string) {
	var agentMode, isFirstContact bool
	err := postgres.DB.QueryRow(
		`SELECT agent_mode, is_first_contact FROM contacts WHERE id = $1`, contactID,
	).Scan(&agentMode, &isFirstContact)
	if err != nil || !agentMode {
		return
	}

	contactType := "returning"
	if isFirstContact {
		contactType = "first_contact"
	}

	var agentPrompt, agentHandoffKeyword string
	err = postgres.DB.QueryRow(`
		SELECT prompt, handoff_keyword
		FROM agents
		WHERE instance_id = $1 AND contact_type = $2 AND is_active = TRUE
	`, inst.ID, contactType).Scan(&agentPrompt, &agentHandoffKeyword)
	if err != nil {
		return // nenhum agente configurado para esta conexão/tipo
	}

	provider, _ := postgres.GetCompanySetting(inst.CompanyID, "ai_provider")
	model, _ := postgres.GetCompanySetting(inst.CompanyID, "ai_model")
	apiKey, _ := postgres.GetCompanySetting(inst.CompanyID, "ai_api_key")
	if provider == "" || model == "" || apiKey == "" {
		log.Printf("[AGENT] Config AI incompleta para empresa %s", inst.CompanyID)
		return
	}

	rows, err := postgres.DB.Query(`
		SELECT direction, content FROM messages
		WHERE contact_id = $1 AND type = 'text'
		ORDER BY created_at DESC LIMIT 20
	`, contactID)
	if err != nil {
		log.Printf("[AGENT] Erro ao carregar histórico: %v", err)
		return
	}
	defer rows.Close()

	var history []aiagent.Message
	for rows.Next() {
		var dir, cnt string
		if err := rows.Scan(&dir, &cnt); err != nil {
			continue
		}
		role := "assistant"
		if dir == "in" {
			role = "user"
		}
		history = append(history, aiagent.Message{Role: role, Content: cnt})
	}
	// Inverter: o DB retorna mais recente primeiro; AI precisa ordem cronológica
	for i, j := 0, len(history)-1; i < j; i, j = i+1, j-1 {
		history[i], history[j] = history[j], history[i]
	}

	cfg := aiagent.Config{Provider: provider, Model: model, APIKey: apiKey}
	aiResp, err := aiagent.Chat(cfg, agentPrompt, history)
	if err != nil {
		log.Printf("[AGENT] Erro na chamada AI: %v", err)
		return
	}

	if strings.Contains(aiResp, agentHandoffKeyword) {
		postgres.DB.Exec(`UPDATE contacts SET agent_mode = FALSE WHERE id = $1`, contactID)
		aiResp = strings.TrimSpace(strings.ReplaceAll(aiResp, agentHandoffKeyword, ""))
		modePayload := map[string]interface{}{
			"event": "agent_mode_changed",
			"data":  map[string]interface{}{"contact_id": contactID, "agent_mode": false},
		}
		modeBytes, _ := json.Marshal(modePayload)
		hub.Global.BroadcastToCompany(inst.CompanyID, modeBytes)
	}

	postgres.DB.Exec(`UPDATE contacts SET is_first_contact = FALSE WHERE id = $1`, contactID)

	if aiResp == "" {
		return
	}

	msgID := uuid.New().String()
	var createdAt string
	err = postgres.DB.QueryRow(
		`INSERT INTO messages (id, instance_id, contact_id, direction, content, type, wa_message_id, media_path, media_name)
		 VALUES ($1, $2, $3, 'out', $4, 'text', '', '', '')
		 RETURNING created_at`,
		msgID, inst.ID, contactID, aiResp,
	).Scan(&createdAt)
	if err != nil {
		log.Printf("[AGENT] Erro ao salvar resposta AI: %v", err)
		return
	}

	outPayload := map[string]interface{}{
		"event": "new_message",
		"data": map[string]interface{}{
			"id":            msgID,
			"instance_id":   inst.ID,
			"instance_name": inst.Name,
			"contact_id":    contactID,
			"phone":         phone,
			"name":          phone,
			"direction":     "out",
			"content":       aiResp,
			"type":          "text",
			"media_path":    "",
			"media_name":    "",
			"created_at":    createdAt,
		},
	}
	outBytes, _ := json.Marshal(outPayload)
	hub.Global.BroadcastToCompany(inst.CompanyID, outBytes)

	if err := service.SendText(inst, phone, aiResp); err != nil {
		log.Printf("[AGENT] Erro ao enviar resposta WhatsApp: %v", err)
	}
}
```

```go
    JOIN instances i ON i.id = c.instance_id
    LEFT JOIN LATERAL (
        SELECT content, direction, type, created_at
        FROM messages
        WHERE contact_id = c.id
        ORDER BY created_at DESC
        LIMIT 1
    ) m ON true
    WHERE m.created_at IS NOT NULL
      AND ($1 = '' OR i.company_id::text = $1)
`
```

```go
func TakeoverConversation(c *gin.Context) {
	instanceName := c.Param("name")
	phone := c.Param("phone")
	companyID := currentCompanyID(c)

	inst, ok := instance.Global.GetByNameScoped(instanceName, companyID)
	if !ok {
		c.JSON(http.StatusNotFound, gin.H{"error": "instância não encontrada"})
		return
	}

	number := strings.TrimPrefix(phone, "+")
	number = strings.ReplaceAll(number, " ", "")

	var contactID string
	if err := postgres.DB.QueryRow(
		`SELECT id FROM contacts WHERE instance_id = $1 AND phone = $2`,
		inst.ID, number,
	).Scan(&contactID); err != nil {
		c.JSON(http.StatusNotFound, gin.H{"error": "contato não encontrado"})
		return
	}

	postgres.DB.Exec(`UPDATE contacts SET agent_mode = FALSE WHERE id = $1`, contactID)

	payload := map[string]interface{}{
		"event": "agent_mode_changed",
		"data":  map[string]interface{}{"contact_id": contactID, "agent_mode": false},
	}
	jsonBytes, _ := json.Marshal(payload)
	hub.Global.BroadcastToCompany(inst.CompanyID, jsonBytes)

	c.JSON(http.StatusOK, gin.H{"ok": true})
}
```

```html
      try {
        const url = this.isEdit ? `/api/agents/${this.form.id}` : '/api/agents';

        const r = await fetch(url, {
          method,
```
